class Solution {
public:
    string longestCommonPrefix(vector<string>& arr) {
        sort(arr.begin(), arr.end());
        string str = "";
        string first = arr[0];
        string last = arr[arr.size() - 1];

        for(int i=0; i<first.length(); i++){
            if(first[i] != last[i] ) break;
            str += first[i];
        }
        return str;
    }
};
